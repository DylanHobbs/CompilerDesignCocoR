Tastier Grammar

Tastier =    "program" ident "{" {VarDecl} {ConstDecl} {ProcDecl} "}"


VarDecl =    Type ident {"," ident} ";"

ConstDecl =    "const" Type ident {"," ident} ";"

ProcDecl =   "void" ident "(" ")"

  "{" {VarDecl} {ConstDecl} {ProcDecl} Stat {Stat} "}"


Type =       "int" | "bool"


Stat =       ident (":=" Expr ";" | "(" ")" ";") {Expr}

           | "if" Expr Stat ["else" Stat]

           | "while Expr Stat

           | "read" ident ";"

           | "write" Expr | string ";"

| "writeln" Expr ";"

           | "{" {Stat | VarDecl | ConstDecl} "}"


Expr =       SimExpr [RelOp SimExpr]

SimExpr =    Term {AddOp Term}

Term =       Primary {MulOp Primary}

Primary =    (ident | number | "-" Primary | "true" 

  | "false" | "(" Expr ")")


AddOp =      "+" | "-"

MulOp =      "*" | "div" | "DIV" | "mod" | "MOD"

RelOp =      "=" | "<" | ">" | "|=" | "<=" | ">="
